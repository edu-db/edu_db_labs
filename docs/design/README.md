## ER-модель
@startuml
class User {
email: TEXT
password: TEXT
name: TEXT
}

class Project {
name: TEXT
description: TEXT
}

class Role {
user.role: (ENUM:"TEAMLEAD", "PROJECT MANAGER", "DEVELOPER")
}


class Action {
actedAt: DATETIME
curent_status: (ENUM:"TO DO", "IN PROGRESS", "RESOLVED", "DONE")
previous_status: (ENUM:"TO DO", "IN PROGRESS", "RESOLVED", "DONE")
}

class Task{
name : TEXT
description: TEXT
dependence: INT
deadline: DATE
status: (ENUM:"TO DO", "IN PROGRESS", "RESOLVED", "DONE")

}

class Artifact{
name: TEXT
description: TEXT
link: TEXT
}

User "(0,*)" -- "(1,1)" Task
User "(1,1)" <--> "(0,*)" Role
Project "(0,*)" <--> "(1,1)" Role
Action "(1,1)" -- "(0,*)" Task
Action "(0,*)" -- "(1,1)" User
Task "(1,1)" -- "(0,*)" Project
Artifact "(0,*)" -- "(1,1)" Project


@enduml
